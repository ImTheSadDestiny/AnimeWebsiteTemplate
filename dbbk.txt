    id int AUTO_INCREMENT,
    nome varchar(20),
    PRIMARY KEY (id)
 );
 
CREATE TABLE serie (
 id int AUTO_INCREMENT,
    nome varchar(40),
    tipo varchar(20),
    trama varchar(1000),
     rilascio varchar(20),
     stato varchar(20),
     voto varchar(20),
     studio varchar(20),
     commenti int,
    durata varchar(20),
    qualit√† varchar(20),
    views int,
    immagine varchar(400),
    PRIMARY KEY (id)
  
);
     

 CREATE TABLE commento (
     id int AUTO_INCREMENT,
     commento varchar(300),
     nome varchar(50),
     data varchar(30),
     fkSerie int,
     PRIMARY KEY (id),
     FOREIGN KEY (fkSerie) REFERENCES serie (id)
     
 );
  CREATE TABLE fkdoppio (
	fkSerie int,
     fkGenere int,
     PRIMARY KEY (fkSerie,fkGenere),
     FOREIGN KEY (fkSerie) REFERENCES serie(id),
     FOREIGN KEY (fkGenere) REFERENCES genere(id)
 );

CREATE TABLE registrati(
    id int AUTO_INCREMENT,
    nomeUtente varchar(40),
    email varchar(50),
    hash varchar(128),
    PRIMARY KEY (id)
);

CREATE TABLE login(
    id int AUTO_INCREMENT,
    nomeUtente varchar(40),
    email varchar(50),
    hash varchar(128),
    PRIMARY KEY (id)
);
CREATE TABLE episodi(
    id int AUTO_INCREMENT,
    numero int,
    fkSerie int,
    video varchar(200),
    PRIMARY KEY (id),
